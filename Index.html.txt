const toggle = document.getElementById('themeToggle');
const root = document.documentElement;
toggle.addEventListener('click', () => {
  const isDark = root.getAttribute('data-theme') === 'dark';
  root.setAttribute('data-theme', isDark ? 'light' : 'dark');

const canvas = document.getElementById('pipeline-canvas');
const ctx = canvas.getContext('2d');
let W, H, nodes = [], particles = [];
function resize() {
  W = canvas.width = window.innerWidth;
  H = canvas.height = window.innerHeight;
}
resize();
window.addEventListener('resize', resize);
function initNodes() {
  nodes = [];
  const cols = Math.ceil(W / 110) + 1;
  const rows = Math.ceil(H / 110) + 1;
  for (let r = 0; r < rows; r++) {
    for (let c = 0; c < cols; c++) {
      if (Math.random() > 0.35) {
        nodes.push({
          x: c * 110 + (r % 2 === 0 ? 0 : 55),
          y: r * 110,
          size: Math.random() * 2 + 1,
          opacity: Math.random() * 0.4 + 0.05,
          pulse: Math.random() * Math.PI * 2
        });
      }
    }
  }
}
initNodes();
window.addEventListener('resize', initNodes);
function spawnParticle() {
  const startX = Math.random() * W;
  particles.push({
    x: startX, y: -10,
    vx: (Math.random() - 0.5) * 0.8,
    vy: Math.random() * 1.2 + 0.5,
    life: 1, decay: Math.random() * 0.003 + 0.002,
    size: Math.random() * 2 + 0.5,
    type: Math.random() > 0.5 ? 'blue' : 'green'
  });
}
let frame = 0;
function animate() {
  ctx.clearRect(0, 0, W, H);
  frame++;
  if (frame % 8 === 0) spawnParticle();
  nodes.forEach((n, i) => {
    n.pulse += 0.02;
    const op = n.opacity * (0.7 + 0.3 * Math.sin(n.pulse));
    ctx.beginPath();
    ctx.arc(n.x, n.y, n.size, 0, Math.PI * 2);
    ctx.fillStyle = `rgba(37,99,235,${op})`;
    ctx.fill();
    nodes.slice(i+1, i+6).forEach(m => {
      const d = Math.hypot(m.x - n.x, m.y - n.y);
      if (d < 150) {
        ctx.beginPath();
        ctx.moveTo(n.x, n.y);
        ctx.lineTo(m.x, m.y);
        ctx.strokeStyle = `rgba(37,99,235,${op * 0.3 * (1 - d/150)})`;
        ctx.lineWidth = 0.5;
        ctx.stroke();
      }
    });
  });
  particles = particles.filter(p => p.life > 0);
  particles.forEach(p => {
    p.x += p.vx; p.y += p.vy;
    p.life -= p.decay;
    const color = p.type === 'blue' ? `rgba(37,99,235,${p.life * 0.6})` : `rgba(34,197,94,${p.life * 0.5})`;
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
    ctx.fillStyle = color;
    ctx.fill();
  });
  requestAnimationFrame(animate);
}
animate();

function animateCounter(el, target, suffix) {
  let start = 0;
  const step = () => {
    start += Math.ceil(target / 60);
    if (start >= target) start = target;
    el.innerHTML = start + '<span class="metric-unit">' + suffix + '</span>';
    if (start < target) requestAnimationFrame(step);
  };
  step();
}
const counterIO = new IntersectionObserver((entries) => {
  entries.forEach(e => {
    if (e.isIntersecting) {
      const nums = e.target.querySelectorAll('.metric-num');
      nums.forEach(n => {
        const txt = n.textContent;
        const suffix = txt.includes('+') ? '+' : txt.includes('%') ? '%' : '';
        const val = parseInt(txt);
        animateCounter(n, val, suffix);
      });
      counterIO.unobserve(e.target);
    }
  });
}, { threshold: 0.3 });
const metricsGrid = document.querySelector('.metrics-grid');
if (metricsGrid) counterIO.observe(metricsGrid);
